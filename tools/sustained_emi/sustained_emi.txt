       , CASE 
           WHEN Ownership_type = 'Joint' THEN SANCTION_AMOUNT / 2
           ELSE SANCTION_AMOUNT
         END AS SANCTION_AMOUNT_new

      ,case when EMI > 1000 and cast(emi as float) / NULLIF(SANCTION_AMOUNT,0) <= 0.25 then emi

      end AS emi1

    when Loan_Type_new = 'Housing Loan' then
        Case
            when dpd_3m > 0  and coalesce(dpd_6m,0) <= 0 and date_opened > DATEADD(month, -5, DATE_TRUNC('month', SCRUB_DATE)) then DATEADD(month, -5, DATE_TRUNC('month', SCRUB_DATE))

            when date_opened >= DATEADD(month, -95, DATE_TRUNC('month', SCRUB_DATE)) then date_opened
            else DATEADD(month, -95, DATE_TRUNC('month', SCRUB_DATE))
        end
    
    when Loan_Type_new = 'Credit Card' then
        Case
            when date_opened >= DATEADD(month, -95, DATE_TRUNC('month', SCRUB_DATE)) then date_opened
            else DATEADD(month, -95, DATE_TRUNC('month', SCRUB_DATE))
        end
    when pay_hist_end_date >= DATEADD(month, -35, DATE_TRUNC('month', SCRUB_DATE)) then pay_hist_end_date 
    else DATEADD(month, -35, DATE_TRUNC('month', SCRUB_DATE)) 
end 
as effective_start_date

    else coalesce(date_closed2,pay_hist_start_date) 
    end

DROP TABLE IF EXISTS sc_buv2_tradeline_data5;

CREATE TABLE sc_buv2_tradeline_data5 AS
WITH base AS (
    SELECT
        t.*,
        crn || '_' ||
        final_emi || '_' ||
        sanction_amount              AS base_fp

    FROM sc_buv2_tradeline_data5_1 t
),

ordered AS (
    SELECT
        b.*,

        LAG(date_closed) OVER (
            PARTITION BY base_fp
            ORDER BY date_opened, effective_end_date desc
        ) AS prev_end

    FROM base b
),

chained AS (
    SELECT
        o.*,

        CASE
            WHEN prev_end IS NULL THEN 1
            WHEN DATEDIFF(month, prev_end, date_opened) BETWEEN 0 AND 6 THEN 0
            ELSE 1
        END AS new_chain_flag

    FROM ordered o
),

chain_numbered AS (
    SELECT
        c.*,
        SUM(new_chain_flag) OVER (
            PARTITION BY base_fp
            ORDER BY date_opened , effective_end_date desc
            ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
        ) AS chain_no

    FROM chained c
)

SELECT
    *,
    base_fp || '_C' || chain_no AS loan_chain_id
FROM chain_numbered;

DROP TABLE IF EXISTS sc_buv2_tradeline_months;
CREATE TABLE sc_buv2_tradeline_months AS
WITH RECURSIVE

params AS (
    SELECT
        6 AS min_months,
        (select distinct scrub_date from sc_buv2_tradeline_data4) as scrub_date,
        DATEADD(month, -95, DATE_TRUNC('month', SCRUB_DATE)) AS window_start_36
),


numbers(n) AS (
    SELECT 1 AS n
    UNION ALL
    SELECT n + 1 FROM numbers WHERE n < 96
),

/* -----------------------------------------------------------
   STEP 2: Expand tradelines into month-level rows
----------------------------------------------------------- */
tradeline_months AS (
    SELECT
        t.crn,
        t.tradeline_id,
        t.non_business_flag,
        DATEADD(
            month,
            n.n,
            DATE_TRUNC('month', t.effective_start_date)
        ) AS month_start,
        Case 
            when t.Loan_Type_new = 'Credit Card' then coalesce(t.CreditLimit , t.High_Credit_Amount) 
            else t.final_emi end as emi_amount,


        Case 
            when t.Loan_Type_new = 'Credit Card' then null 
            else cast(t.FINAL_EMI_ADJ as bigint) end as emi_amount_adj    
            
    FROM sc_buv2_tradeline_data5 t
    JOIN numbers n
      ON DATEADD(
             month,
             n.n,
             DATE_TRUNC('month', t.effective_start_date)
         )
         BETWEEN
             (SELECT window_start_36 FROM params)
             AND DATE_TRUNC('month', scrub_date )
        AND 
        DATEADD(
             month,
             n.n,
             DATE_TRUNC('month', t.effective_start_date )
         ) <= t.effective_end_date
)

SELECT * FROM tradeline_months;

        cc.crn,
        cc.month_start,
        COALESCE(SUM(Case when non_business_flag = 1 then tm.emi_amount end ), 0) AS total_emi,
        COALESCE(SUM(Case when non_business_flag = 1 then tm.emi_amount_adj end), 0) AS total_emi_adj,
        COALESCE(SUM(Case when non_business_flag = 0 then tm.emi_amount end ), 0) AS total_emi_bl,
        COALESCE(SUM(Case when non_business_flag = 0 then tm.emi_amount_adj end), 0) AS total_emi_adj_bl,

        COALESCE(max(Case when non_business_flag = 2 then tm.emi_amount end ), 0) AS max_cc_limit
    
    FROM customer_calendar cc
    LEFT JOIN sc_buv2_tradeline_months tm
      ON cc.crn = tm.crn
     AND cc.month_start = tm.month_start
    GROUP BY cc.crn, cc.month_start
)

SELECT * FROM customer_month_emi;

DROP TABLE IF EXISTS sc_buv2_tradeline_rolling_windows;
CREATE TABLE sc_buv2_tradeline_rolling_windows AS
WITH 

params AS (
    SELECT
        6 AS min_months,
        (select distinct scrub_date from sc_buv2_tradeline_data4) as scrub_date,
        DATEADD(month, -95, DATE_TRUNC('month', SCRUB_DATE)) AS window_start_36
),


rolling_windows AS (
    SELECT
        crn,
        month_start AS window_end,
        DATEADD(
            month,
            - (SELECT min_months FROM params) + 1,
            month_start
        ) AS window_start,
        MIN(total_emi) OVER (
            PARTITION BY crn
            ORDER BY month_start
            ROWS BETWEEN 5 PRECEDING -- change as per min month window
                AND CURRENT ROW
        ) AS sustained_emi,
        MIN(total_emi_adj) OVER (
            PARTITION BY crn
            ORDER BY month_start
            ROWS BETWEEN 5 PRECEDING -- change as per min month window
                AND CURRENT ROW
        ) AS sustained_emi_adj,

        MIN(total_emi_bl) OVER (
            PARTITION BY crn
            ORDER BY month_start
            ROWS BETWEEN 5 PRECEDING -- change as per min month window
                AND CURRENT ROW
        ) AS sustained_emi_bl,

        MIN(total_emi_adj_bl) OVER (
            PARTITION BY crn
            ORDER BY month_start
            ROWS BETWEEN 5 PRECEDING -- change as per min month window
                AND CURRENT ROW
        ) AS sustained_emi_adj_bl,

        MAX(max_cc_limit) OVER (
            PARTITION BY crn,month_start
        ) AS max_cc_limit

    FROM sc_buv2_tradeline_months_sum tm
),

valid_windows AS (
    SELECT *
    FROM rolling_windows
    WHERE window_start >= (SELECT window_start_36 FROM params)
)

SELECT * FROM valid_windows;

DROP TABLE IF EXISTS sc_buv2_tradeline_best_rolling_window;
CREATE TABLE sc_buv2_tradeline_best_rolling_window AS
WITH 

best_window AS (
    SELECT
        *,
        ROW_NUMBER() OVER (
            PARTITION BY crn
            ORDER BY sustained_emi DESC, window_start ASC
        ) AS rn,

        ROW_NUMBER() OVER (
            PARTITION BY crn
            ORDER BY sustained_emi_bl DESC, window_start ASC
        ) AS rn_bl,

        ROW_NUMBER() OVER (
            PARTITION BY crn
            ORDER BY max_cc_limit DESC, window_start ASC
        ) AS rn_cc

    FROM sc_buv2_tradeline_rolling_windows
)

SELECT A.CRN 
,B.window_start
,B.window_end
,B.sustained_emi
,B.sustained_emi_adj

,C.window_start as window_start_bl
,C.window_end as window_end_bl
,C.sustained_emi_bl
,C.sustained_emi_adj_bl

,D.window_end as cc_month
,D.max_cc_limit

FROM (SELECT DISTINCT CRN FROM best_window) A
LEFT JOIN best_window B ON A.CRN = B.CRN AND B.RN = 1
LEFT JOIN best_window C ON A.CRN = C.CRN AND C.RN_BL = 1
LEFT JOIN best_window D ON A.CRN = D.CRN AND D.RN_CC = 1;

    WHEN cc_month >= '2022-01-01' THEN CC_CL_MAX_AFFL
    WHEN max_cc_limit BETWEEN 10000 AND 29999 THEN 10000
    WHEN max_cc_limit BETWEEN 30000 AND 49949 THEN 20000

    WHEN max_cc_limit BETWEEN 229946 AND 332999 THEN 65699
    WHEN max_cc_limit BETWEEN 333000 AND 4129999 THEN 80775
    ELSE 0
END AS CC_CL_min_AFFL

FROM sc_buv2_tradeline_best_rolling_window
where max_cc_limit > 0)

SELECT * 
,GREATEST(CC_CL_MAX_AFFL,CC_CL_min_AFFL) AS CC_INCOME

FROM SUB;
